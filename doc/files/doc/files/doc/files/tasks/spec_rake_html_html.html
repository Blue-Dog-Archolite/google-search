require &amp;8216;spec/rake/spectask&amp;8216;

desc &amp;8220;Run all specifications&amp;8221; Spec::Rake::SpecTask.new(:spec) do |t| 

 
 


  t.libs &amp;lt;&amp;lt; &amp;quot;lib&amp;quot;
  t.spec_opts = [&amp;quot;--color&amp;quot;, &amp;quot;--require&amp;quot;, &amp;quot;spec/spec_helper.rb&amp;quot;]

end

namespace :spec do 

 
 


  desc &amp;quot;Run all specifications verbosely&amp;quot;
  Spec::Rake::SpecTask.new(:verbose) do |t|
    t.libs &amp;lt;&amp;lt; &amp;quot;lib&amp;quot;
    t.spec_opts = [&amp;quot;--color&amp;quot;, &amp;quot;--format&amp;quot;, &amp;quot;specdoc&amp;quot;, &amp;quot;--require&amp;quot;, &amp;quot;spec/spec_helper.rb&amp;quot;]
  end

  desc &amp;quot;Run specific specification verbosely (specify SPEC)&amp;quot;
  Spec::Rake::SpecTask.new(:select) do |t|
    t.libs &amp;lt;&amp;lt; &amp;quot;lib&amp;quot;
    t.spec_files = [ENV[&amp;quot;SPEC&amp;quot;]]
    t.spec_opts = [&amp;quot;--color&amp;quot;, &amp;quot;--format&amp;quot;, &amp;quot;specdoc&amp;quot;, &amp;quot;--require&amp;quot;, &amp;quot;spec/spec_helper.rb&amp;quot;]
  end

end
