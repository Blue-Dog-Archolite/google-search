require &8216;spec/rake/spectask&8216;

desc &8220;Run all specifications&8221; Spec::Rake::SpecTask.new(:spec) do |t| 



  t.libs &lt;&lt; &quot;lib&quot;
  t.spec_opts = [&quot;--color&quot;, &quot;--require&quot;, &quot;spec/spec_helper.rb&quot;]

end

namespace :spec do 



  desc &quot;Run all specifications verbosely&quot;
  Spec::Rake::SpecTask.new(:verbose) do |t|
    t.libs &lt;&lt; &quot;lib&quot;
    t.spec_opts = [&quot;--color&quot;, &quot;--format&quot;, &quot;specdoc&quot;, &quot;--require&quot;, &quot;spec/spec_helper.rb&quot;]
  end

  desc &quot;Run specific specification verbosely (specify SPEC)&quot;
  Spec::Rake::SpecTask.new(:select) do |t|
    t.libs &lt;&lt; &quot;lib&quot;
    t.spec_files = [ENV[&quot;SPEC&quot;]]
    t.spec_opts = [&quot;--color&quot;, &quot;--format&quot;, &quot;specdoc&quot;, &quot;--require&quot;, &quot;spec/spec_helper.rb&quot;]
  end

end
