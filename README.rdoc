== Example Google::Search::Item::Web

  item.title         # => 'foobar - Wikipedia, the free encyclopedia'
  item.uri           # => 'http://en.wikipedia.org/wiki/Foobar' 
  item.cache _uri    # => 'http://www.google.com/search?q=cache:4styY9qq7tYJ:en.wikipedia.org'
  item.unescaped_uri # => 'http://en.wikipedia.org/wiki/Foobar'

  item.content       # => 'Foobar is often used....'
